<section id="conditions">
    <h2>Conditions idéales</h2>
    <ul>
        <li>Plein soleil, minimum 6h/jour</li>
        <li>Sol léger et bien drainé</li>
        <li>Arrosage modéré</li>
        <li>Espacement 20-30 cm</li>
        <li>Semis idéal au printemps</li>
    </ul>
